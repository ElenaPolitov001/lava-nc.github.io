PilotNet SNN Example

Network excange module is available as lava.lib.dl.netx.{hdf5, blocks, utils}. * hdf5 implements automatic network generation. * blocks implements individual layer blocks. * utils implements hdf5 reading utilities.

Create network block

A network block can be created by simply instantiating netx.hdf5.Network with the path of the desired hdf5 network description file. * The input layer is accessible as net.in_layer. * The output layer is accessible as net.out_layer. * All the constituent layers are accessible as as a list: net.layers.

from lava.magma.core.resources import CPU
from lava.magma.core.decorator import implements, requires
from lava.magma.core.sync.protocols.loihi_protocol import LoihiProtocol
from lava.magma.core.model.py.type import LavaPyType
from lava.magma.core.model.py.ports import PyRefPort

class ReadAndReset(io.sink.Read, io.reset.Reset):
    def __init__(
        self,
        buffer: int,
        interval: int = 1,
        offset: int = 0,
        reset_value: int = 0,
    ) -> None:
        super(io.sink.Read, self).__init__(buffer, interval, offset)
        # print(self.buffer)
        print(self.interval)
        super(io.reset.Reset, self).__init__(reset_value, interval, offset)

    def connect_var(self, var: Var) -> None:
        super(io.sink.Read, self).connect_var(var)

@implements(proc=ReadAndReset, protocol=LoihiProtocol)
@requires(CPU)
class PyReadAndReset(io.sink.AbstractPyRead, io.reset.AbstractPyReset):
    state: PyRefPort = None
    data = None
    interval: np.ndarray = LavaPyType(np.ndarray, int)
    offset: np.ndarray = LavaPyType(np.ndarray, int)
    reset_value: np.ndarray = LavaPyType(np.ndarray, int)

    def run_post_mgmt(self) -> None:
        super(io.sink.AbstractPyRead, self).run_post_mgmt()
        super(io.reset.AbstractPyReset, self).run_post_mgmt()
